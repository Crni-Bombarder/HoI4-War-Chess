# Challenge refused
country_event = {
    id = wchess_events.2
    title = wchess_events.2.t
    desc = wchess_events.2.desc

    is_triggered_only = yes
    timeout_days = 5

    option = {
        name = wchess_events.2.a
        var:wchess_challenged = {
            clear_variable = THIS.wchess_challenger
        }
        clear_variable = THIS.wchess_challenged
    }
    option = {
        log = "[ROOT.GetName]"
        name = wchess_events.2.b
        every_other_country = {
            if = {
                limit = {
                    check_variable = {
                        var = THIS.wchess_challenged
                        value = PREV
                        compare = equals
                    }
                }
                country_event = wchess_events.4
            } else = {
                set_variable = {
                    var = THIS.wchess_shamed
                    value = ROOT.wchess_challenger
                }
                set_variable = {
                    var = THIS.wchess_has_shamed
                    value = ROOT
                }
                country_event = wchess_events.3
            }
        }

        country_event = wchess_events.5
    }
}
