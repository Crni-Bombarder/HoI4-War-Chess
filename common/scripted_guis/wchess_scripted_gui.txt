@B_KING   = 1
@W_KING   = 2
@B_QUEEN  = 3
@W_QUEEN  = 4
@B_BISHOP = 5
@W_BISHOP = 6
@B_ROOK   = 7
@W_ROOK   = 8
@B_KNIGHT = 9
@W_KNIGHT = 10
@B_PAWN   = 11
@W_PAWN   = 12
@EMPTY    = 13

@ID_KING   = 0
@ID_QUEEN  = 1
@ID_BISHOP = 2
@ID_ROOK   = 3
@ID_KNIGHT = 4
@ID_PAWN   = 5

@BLACK = 1
@WHITE = 0

@ACTIVATE = 1
@DEACTIVATE = 2

@S_SELECTION = 0

scripted_gui = {
    wchess_scripted_gui_board = {
        window_name = "wchess_gui_board"

        context_type = player_context

        visible = {
            has_variable = wchess_start_game
        }

        dynamic_lists = {
            wchess_grid_pieces = {
                array = wchess_pieces_array
                change_scope = no
                entry_container = wchess_piece_cell
            }

            wchess_grid_highlights = {
                array = wchess_highlights_array
                change_scope = no
                entry_container = wchess_highlight_cell
            }

            wchess_grid_halochecks = {
                array = wchess_halochecks_array
                change_scope = no
                entry_container = wchess_halocheck_cell
            }
        }

        effects = {
            wchess_piece_tile_click = {
                log = "Event click on [?i], piece [?v] [?wchess_play_state]"
                if = {
                    limit = {
                        has_variable = wchess_game_turn
                    }

                    # State : Piece / Pawn selection
                    if = {
                        limit = {
                            check_variable = {
                                wchess_play_state = @S_SELECTION
                            }
                            check_variable = {
                                var = wchess_pieces_array^i
                                value = @EMPTY
                                compare = not_equals
                            }
                            set_temp_variable = {
                                wchess_temp_check_color_piece = wchess_pieces_array^i
                            }
                            modulo_temp_variable = {
                                wchess_temp_check_color_piece = 2
                            }
                            check_variable = {
                                wchess_temp_check_color_piece = wchess_color
                            }
                        }

                        if = {
                            limit = {
                                check_variable = {wchess_highlights_array^i = @DEACTIVATE}
                            }
                            set_variable = {wchess_highlights_array^i = @ACTIVATE}
                        } else = {
                            set_variable = {wchess_highlights_array^i = @DEACTIVATE}
                        }
                    }
                }
            }
        }

        properties = {
            wchess_piece_tile = {
                frame = v
            }

            wchess_highlight_tile = {
                frame = v
            }

            wchess_halocheck_tile = {
                frame = v
            }
        }
    }

    # ########################################## #
    # ################# DEBUG ################## #
    # ########################################## #

    wchess_scripted_gui_debug = {
        window_name = "wchess_debug_interface"

        context_type = player_context

        visible = {
            has_variable = wchess_start_game
        }

        effects = {

            wchess_debug_random_tile_array_click = {
                log = "Random value to array wchess_pieces_array"
                resize_array = {
                    array = wchess_pieces_array
                    size = 64
                    value = 2
                }
                for_each_loop = {
                    array = wchess_pieces_array
                    index = i
                    set_temp_variable = {temp_random = random}
                    multiply_temp_variable = {
                        var = temp_random
                        value = 13.5
                    }
                    round_temp_variable = temp_random
                    clamp_temp_variable = {
                        var = temp_random
                        min = 0
                        max = 13
                    }

                    set_variable = {wchess_pieces_array^i = temp_random}
                }
            }

            wchess_debug_clean_tile_array_click = {
                log = "Clearing wchess_pieces_array"
                clear_array = wchess_pieces_array
            }

            wchess_debug_add_highlight_array_click = {
                log = "Random value to array wchess_highlights_array"
                clear_array = wchess_highlights_array
                resize_array = {
                    array = wchess_highlights_array
                    size = 64
                    value = 1
                }
                for_each_loop = {
                    array = wchess_highlights_array
                    index = i
                    if = {
                        limit = {
                            check_variable = {
                                random < 0.5
                            }
                        }
                        set_variable = {wchess_highlights_array^i = 2}
                    }
                    log = "[?wchess_highlights_array^i]"
                }
            }

            wchess_debug_clean_highlight_array_click = {
                log = "Clearing wchess_highlights_array"
                clear_array = wchess_highlights_array
            }

            wchess_debug_add_halocheck_array_click = {
                log = "Random value to array wchess_halochecks_array"
                clear_array = wchess_halochecks_array
                resize_array = {
                    array = wchess_halochecks_array
                    size = 64
                    value = 2
                }
                for_each_loop = {
                    array = wchess_pieces_array
                    index = i
                    value = piece
                    if = {
                        limit = {
                            check_variable = {
                                random < 0.5
                            }
                            check_variable = {
                                piece < 3
                            }
                        }
                        set_variable = {wchess_halochecks_array^i = 1}
                    }
                    log = "[?wchess_halochecks_array^i] [?piece]"
                }
            }

            wchess_debug_clean_halocheck_array_click = {
                log = "Clearing wchess_halochecks_array"
                clear_array = wchess_halochecks_array
            }
        }
    }
}
